    function log(message){
      out.textContent = `${new Date().toLocaleTimeString()} · ${message}

    async function writeTest(id = "debug-test"){
      try{
        const ref = doc(db, "articleStats", id);
        log(`Leyendo articleStats/${id}...`);
        const snap = await getDoc(ref);
        if(snap.exists()){
          const current = Number(snap.data()?.views || 0);
          const next = current + 1;
          log(`Existe. views actuales=${current}. Actualizando a ${next}...`);
          await updateDoc(ref, { views: next });
        }else{
          log("No existe. Creando documento con views=1...");
          await setDoc(ref, { views: 1 });
        }
        const after = await getDoc(ref);
        log(`✅ OK. Documento guardado. views=${Number(after.data()?.views || 0)}`);
      }catch(error){
        log(`❌ ERROR: ${error.code || error.name || "unknown"} · ${error.message || error}`);
        console.error(error);
      }
    }

    document.getElementById("btn").onclick = () => writeTest("debug-test");
    document.getElementById("btnRandom").onclick = () => writeTest(`debug-${Date.now()}`);
    log("Firebase inicializado. Listo para probar.");
    writeTest("debug-test");
